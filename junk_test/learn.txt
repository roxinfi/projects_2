
#include "mcc_generated_files/mcc.h"
#define SWITCH_ON           0
#define RED_LEDS            LATB
#define SET_SWITCH_1_PUSH   SET_SWITCH_1_GetValue()
#define SET_SWITCH_2_PUSH   SET_SWITCH_2_GetValue()
#define SET_SWITCH_3_PUSH   SET_SWITCH_3_GetValue()

#define GREEN_LED_ON        GREEN_LED_SetHigh();
#define GREEN_LED_OFF       GREEN_LED_SetLow();


void main(void)
{
    SYSTEM_Initialize();
    char leds ;
    leds = 0;
    char counter;
    counter = 0; 
    char y = 0x00;

    while (1)
    {
        if(SET_SWITCH_1_GetValue() == SWITCH_ON)
            {
            YELLOW_LED_Toggle();
            }
        if (SET_SWITCH_2_GetValue() == SWITCH_ON)
            {
            GREEN_LED_SetHigh();
            }
        else
            {
            GREEN_LED_SetLow();
            }
        if (SET_SWITCH_3_GetValue() == SWITCH_ON)
            {
            RED_LEDS = (leds);
            }
        else
            {
            if (counter >= 2 )
                {
                y = ~(y);
                y = ((y)&(0x02));
                RED_LEDS = ~(y);
                counter = 0;
                }
            counter++;

            }
        __delay_ms(100);
    }
}

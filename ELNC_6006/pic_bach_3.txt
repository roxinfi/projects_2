void pattern1(void)
{
    LED_3_Toggle();

void pattern2(void)
{

}

{
    SYSTEM_Initialize();

    {
        __delay_ms(250);
        if(SWITCH_2_GetValue()==SWITCH_ON)
        {
            YELLOW_LED_SetHigh();
        }
        else
        {
            YELLOW_LED_SetLow();
        }
        if(SWITCH_3_GetValue()==SWITCH_ON)
        {
            GREEN_LED_Toggle();
        }
        if ((SWITCH_3_GetValue()==SWITCH_ON)&&(SWITCH_2_GetValue()==SWITCH_ON))
        {
            if(counter == 6)
            {
                pattern1();
                //logic = (logic)&(0x08);
                //LEDS = logic;
                //logic = ~(logic);
                //logic = (logic)|(0x07);
                counter = 0;
            }

        else
            {
            pattern2();
            }
    }
}
